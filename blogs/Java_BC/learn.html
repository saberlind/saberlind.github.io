    @RequestMapping(value = "/printDispatch", method = RequestMethod.GET)
    public String printDispatch(@RequestParam String dispatchSn, HttpServletResponse response) {
        TmDispatchEntity tmDispatchEntity = tmDispatchService.selectOne(EntityWrapperUtil.createNeDelete(TmDispatchEntity.class).eq("DISPATCH_SN", dispatchSn));
        JasperPrint print;
        try (ServletOutputStream ouputStream = response.getOutputStream(); Connection connection = dataSource.getConnection()) {
            Map<String, Object> map = this.putParams(tmDispatchEntity);
            // String pdfTempletPath = FileUtils.getRootPath() + "/reports/gtms_dispatch.jasper";
            String pdfTempletPath = "/home/jboss/reports/gtms_dispatch.jasper";
            print = JasperFillManager.fillReport(pdfTempletPath, map,connection);
            byte[] bytes = JasperExportManager.exportReportToPdf(print);
            response.setHeader("Content-Disposition", "attachment;filename=" + dispatchSn + "_gtms_dispatch.pdf");
            response.setContentType("application/pdf");
            response.setContentLength(bytes.length);
            ouputStream.write(bytes, 0, bytes.length);
            ouputStream.flush();
        } catch (Exception e) {
            log.error("print,error:" + e);
        }
        return "";
    }

    /**
 * JDBC sink batch options.
 */
@PublicEvolving
public class JdbcExecutionOptions implements Serializable {
	public static final int DEFAULT_MAX_RETRY_TIMES = 3;
	private static final int DEFAULT_INTERVAL_MILLIS = 0;
	public static final int DEFAULT_SIZE = 5000;

	private final long batchIntervalMs;
	private final int batchSize;
	private final int maxRetries;

	private JdbcExecutionOptions(long batchIntervalMs, int batchSize, int maxRetries) {
		Preconditions.checkArgument(maxRetries >= 0);
		this.batchIntervalMs = batchIntervalMs;
		this.batchSize = batchSize;
		this.maxRetries = maxRetries;
	}

	public long getBatchIntervalMs() {
		return batchIntervalMs;
	}

	public int getBatchSize() {
		return batchSize;
	}

	public int getMaxRetries() {
		return maxRetries;
	}

	@Override
	public boolean equals(Object o) {
		if (this == o) {
			return true;
		}
		if (o == null || getClass() != o.getClass()) {
			return false;
		}
		JdbcExecutionOptions that = (JdbcExecutionOptions) o;
		return batchIntervalMs == that.batchIntervalMs &&
			batchSize == that.batchSize &&
			maxRetries == that.maxRetries;
	}

	@Override
	public int hashCode() {
		return Objects.hash(batchIntervalMs, batchSize, maxRetries);
	}

	public static Builder builder() {
		return new Builder();
	}

	public static JdbcExecutionOptions defaults() {
		return builder().build();
	}

	/**
	 * Builder for {@link JdbcExecutionOptions}.
	 */
	public static final class Builder {
		private long intervalMs = DEFAULT_INTERVAL_MILLIS;
		private int size = DEFAULT_SIZE;
		private int maxRetries = DEFAULT_MAX_RETRY_TIMES;

		public Builder withBatchSize(int size) {
			this.size = size;
			return this;
		}

		public Builder withBatchIntervalMs(long intervalMs) {
			this.intervalMs = intervalMs;
			return this;
		}

		public Builder withMaxRetries(int maxRetries) {
			this.maxRetries = maxRetries;
			return this;
		}

		public JdbcExecutionOptions build() {
			return new JdbcExecutionOptions(intervalMs, size, maxRetries);
		}
	}
}

    # RedisNumberUtil

    @Component
@Slf4j
public class TmRedisNumberUtil implements CommandLineRunner {

    private static RedisServiceFacade redisService;
    private static SysSerialConfigService sysSerialConfigService;
    private static SysGlobalConfigService sysGlobalConfigService;
    private static SysDictionaryService sysDictionaryService;

    @Override
    public void run(String... args) throws Exception {
        this.init();
    }

    public void init() {
        log.info("TmRedisNumberUtil init ...");
        redisService = SpringContextHelper.getBean(RedisServiceFacade.class);
        sysSerialConfigService = SpringContextHelper.getBean(SysSerialConfigService.class);
        sysGlobalConfigService = SpringContextHelper.getBean(SysGlobalConfigService.class);
        sysDictionaryService = SpringContextHelper.getBean(SysDictionaryService.class);
        log.info("TmRedisNumberUtil init success");
    }

    /**
     * 前缀 + 时间key(自定义时间格式) + 唯一随机数（自定义位数）
     * @param type 类型,取货计划、运输计划、应收/应付暂估、应收/应付结算、索赔、事件等
     * @param prefix 前缀
     * @param dateFormat 日期
     * @param randomLength 随机数位数
     * @return
     */
    public static String createNumber(String type, String prefix, String dateFormat, Long randomLength) {
        if (StringUtils.isBlank(type)) {
            throw new BizException("type不能为空");
        }
        StringBuilder sb = new StringBuilder();
        if (StringUtils.isNotBlank(prefix)) {
            sb.append(prefix);
        }
        if (StringUtils.isNotBlank(dateFormat)) {
            String key = new SimpleDateFormat(dateFormat).format(new Date());
            sb.append(key);
        }
        if (null != randomLength) {
            String random = String.format("%0" + randomLength + "d", getRedisRandomNumber(type, sb.toString()));
            sb.append(random);
        }
        return sb.toString();
    }

    /**
     * 获得唯一随机数据
     * @param type
     * @param key
     * @return
     */
    public static Long getRedisRandomNumber(String type, String key) {
        boolean result = false;
        key = "factorySn_" + key;
        RedisDistributedCounterObject range = new RedisDistributedCounterObject(key);
        try {
            // key的值自增1
            long increase = redisService.increase(range, 1);
            log.info("transportSn ---》key: {}, value: {}", key, increase);
            // 获取自增梯度全局配置
            String configValue = sysGlobalConfigService.getInfo("tms.sys.sn.increase.gradient", "20");
            if (1 == increase) {
                // 需要处理异常情况，例如redis服务异常、redis集群升级
                // 获取当前号段最大值
                SysSerialConfigEntity serialConfigEntity = sysSerialConfigService.getByType(type);
                if (null == serialConfigEntity) {
                    // 初始化配置
                    serialConfigEntity.setType(type);
                    serialConfigEntity.setIname(sysDictionaryService.queryRedisBySn(type));
                    serialConfigEntity.setValidDate(DateUtils.parseDate(DateUtils.format(new Date(), DATE_FORMAT_STRING), DATE_FORMAT_STRING));
                    serialConfigEntity.setSegmentMax(20L);
                    serialConfigEntity.setAddUserId(null != LoginUserUtil.getId() ? LoginUserUtil.getId() : null);
                    serialConfigEntity.setAddTime(new Date());
                    serialConfigEntity.setState(StatusEnum.ENABLED.getValue());
                    serialConfigEntity.setCurrentVersion(1L);
                    serialConfigEntity = new SysSerialConfigEntity();
                    sysSerialConfigService.insert(serialConfigEntity);
                    return increase;
                }else {
                    // 当天配置已初始化，说明号码1已取号，但获取的redis序号为1，是异常情况
                    return updateRedisAndSegment(key, configValue, serialConfigEntity);
                }
            } else {
                // 不等于1,说明未发生异常情况,自增后的当前序号对梯度取余,余数为0则说明号段最大值需要按梯度自增
                if (increase % Long.parseLong(configValue) == 0) {
                    // 获取当前号段最大值
                    SysSerialConfigEntity serialConfigEntity = sysSerialConfigService.getByType(type);
                    if (null == serialConfigEntity) {
                        // 新增配置
                        serialConfigEntity = new SysSerialConfigEntity();
                        // 例如当前increase为configValue,segmentMax需要配置为(1+1)*configValue
                        Long segmentMax = ((increase / Long.parseLong(configValue)) + 1) * Long.parseLong(configValue);
                        serialConfigEntity.setType(type);
                        serialConfigEntity.setIname(sysDictionaryService.queryRedisBySn(type));
                        serialConfigEntity.setValidDate(DateUtils.parseDate(DateUtils.format(new Date(), DATE_FORMAT_STRING), DATE_FORMAT_STRING));
                        serialConfigEntity.setSegmentMax(segmentMax);
                        serialConfigEntity.setAddUserId(null != LoginUserUtil.getId() ? LoginUserUtil.getId() : null);
                        serialConfigEntity.setAddTime(new Date());
                        serialConfigEntity.setState(StatusEnum.ENABLED.getValue());
                        serialConfigEntity.setCurrentVersion(1L);
                        sysSerialConfigService.insert(serialConfigEntity);
                    } else {
                        // 更新配置
                        serialConfigEntity.setSegmentMax(serialConfigEntity.getSegmentMax() + Long.parseLong(configValue));
                        Long currentVersion = serialConfigEntity.getCurrentVersion();
                        serialConfigEntity.setCurrentVersion(serialConfigEntity.getCurrentVersion() + 1);
                        result = sysSerialConfigService.update(serialConfigEntity, EntityWrapperUtil.create().eq("TYPE", type).eq("CURRENT_VERSION",currentVersion));
                        if (!result) {
                            throw new BizException("更新序号配置失败,currentVersion=" + currentVersion);
                        }
                    }
                }
                return increase;
            }
        } catch (Exception e) {
            log.error("获取序号失败:",e);
            throw new BizException("获取序号失败");
        }
    }

    private static Long updateRedisAndSegment(String key,String configValue,SysSerialConfigEntity serialConfigEntity) {
        boolean result = false;
        // 否则是异常情况,设置缓存计数初始值为号段最大值+1
        Long currentValue = serialConfigEntity.getSegmentMax() + 1;
        redisService.set(key, currentValue);
        // 并梯度自增更新号段最大值
        serialConfigEntity.setSegmentMax(serialConfigEntity.getSegmentMax() + Long.parseLong(configValue));
        Long currentVersion = serialConfigEntity.getCurrentVersion();
        serialConfigEntity.setCurrentVersion(serialConfigEntity.getCurrentVersion() + 1);
        result = sysSerialConfigService.update(serialConfigEntity, EntityWrapperUtil.create().eq("TYPE", serialConfigEntity.getType()).eq("CURRENT_VERSION",currentVersion));
        if (!result) {
            throw new BizException("更新序号配置失败,currentVersion=" + currentVersion);
        }
        return currentValue;
    }

}

public class ActionMap implements InitializingBean {

    @Autowired

                log.error("未定义类型");
            }
        });
    }

    public OrderBusinessAbstract get(String key) {
        return ORDER_BUSINESS_ABSTRACT_MAP.get(key);
    }

}

    # 缓存

    注解 说明
    @EnableCaching 开启缓存注解功能
    @Cacheable 在方法执行前spring先查看缓存中是否有数据，如果有数据，则直接返回缓存数据若没有数据，调用方法并将方法返回值放到缓存中,一般使用在查询方法中
    @CachePut 将方法的返回值放到缓存中，一般使用在添加方法中
    @CacheEvict 将一条或多条数据从缓存中删除，一般使用在删除和更新方法中

    @Cacheable部分参数：（ value：缓存的名称，每个缓存名称下面可以有多个key key：缓存的key condition：条件，满足该条件是才缓存数据 unless:满足条件则不缓存）

    @CachePut部分参数：（ value：缓存的名称，每个缓存名称下面可以有多个key key：缓存的key）

    @CacheEvict部分参数：（ value：缓存的名称，每个缓存名称下面可以有多个key key：缓存的key）

    <?xml version="1.0" encoding="UTF-8"?>
<configuration>
    <!--<springProperty scope="context" name="spring.application.name" source="spring.application.name"/>-->
    <springProperty scope="context" name="theLogPath" source="logging.file.path"/>
    <!-- 文件输出格式 -->
    <property name="PATTERN" value="%-12(%d{yyyy-MM-dd HH:mm:ss.SSS}) |-%-5level [%X{X-Request-Id:--}] [%thread] %c [%L] -| %msg%n" />

    <property name="spring.application.name" value="business-frequency-adjust-server" />
    <property name="FILE_PATH" value="${theLogPath}" />

    <appender name="CONSOLE-APPENDER" class="ch.qos.logback.core.ConsoleAppender">
        <encoder>
            <pattern>${PATTERN}</pattern>
            <charset>UTF-8</charset>
        </encoder>
    </appender>

    <appender name="FILE-APPENDER" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <!-- 文件路径 -->
        <file>${FILE_PATH}/${spring.application.name}.log</file>
        <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
            <!-- 文件名称 -->
            <fileNamePattern>${FILE_PATH}/${spring.application.name}%d{yyyy-MM-dd}%i.log</fileNamePattern>
            <!-- 文件最大保存历史数量 -->
            <maxFileSize>100MB</maxFileSize>
            <MaxHistory>100</MaxHistory>
            <totalSizeCap>10GB</totalSizeCap>
            <cleanHistoryOnStart>true</cleanHistoryOnStart>
        </rollingPolicy>
        <append>true</append>
        <layout class="ch.qos.logback.classic.PatternLayout">
            <pattern>${PATTERN}</pattern>
        </layout>
    </appender>
