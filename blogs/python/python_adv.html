# 求多个数的平均值
nums =[50,30,20,10,77]
print(sum(nums)/len(nums))


# nums = [] 将nums定义在全局中，容易被其他地方修改
# 形成闭包的条件：
#   1.函数嵌套
#   2.将内部函数作为返回值返回
#   3.内部函数必须要使用到外部函数的变量
def avg():

    nums = [] #将nums定义在函数中，可以保证数据的安全

    def inner(n):
        nums.append(n)
        return sum(nums) / len(nums)

    return inner

avg = avg()

print(avg(10))
print(avg(10))
nums = ['hello']
print(avg(30))
print(avg(50))
print(avg(10))

# 以上方式对函数进行扩展比较繁琐

# 以下是装饰器实现方式
def begin_end(old):
    """
        用来对其他函数进行扩展，使其他函数可以在执行前和执行后打印指定内容
    :return: new_function
    """
    def new_function(*args, **kwargs):
        print('开始执行~~~~~')
        # 调用被扩展的函数
        result = old(*args, **kwargs)
        print('执行结束~~~~~')
        return result
    # 返回新函数
    return new_function

f = begin_end(fn)
f1 = begin_end(add)
f2 = begin_end(mul)

# r = f()
# r = f1(123,456)
r = f2(123,456)

print(r)

    # 如何隐藏一个对象中的属性？
#   将对象的属性名，修改为一个外部不知道的名字

class Person:
    """
        表示Person类
    """
    def __init__(self, name):
        self.hidden_name = name

    def get_name (self):
        return self.hidden_name

    def set_name(self, name):
        self.hidden_name = name

# 可以为对象的属性使用双下划线开头，__xxx
# 双下划线开头的属性，是对象的隐藏属性，隐藏属性只能在类的内部访问，无法通过对象访问
# 其实隐藏属性只不过是Python自动为属性改了一个名字
#   实际上是将名字修改为了： _类名__属性名  比如 __name -> _Dog__name
class Dog:
    """
        定义Dog类
    """
    def __init__(self, name):
        self.__name = name

    @property
    def name(self):
        return self.__name

    @name.setter
    def name(self, name):
        self.__name = name

d = Dog("金毛")
d._Dog__name = "泰迪"
print(d._Dog__name)
d.name = "比熊"
print(d.name)

    # 继承

    # 在创建类时，如果省略了父类，则默认父类为 object
class Animal:
    def __init__(self, name):
        self._name = name

    @property
    def name(self):
        return self._name

    @name.setter
    def name(self, name):
        self._name = name


    def run(self):
        print("动物跑~~~")

    def sleep(self):
        print("动物睡觉~~~")

class Dog(Animal):

    # def __init__(self, name, age):
    #     self.name = name
    #     self.age = age
    def __init__(self, name, age):
        # Animal.__init__(self,name)
        super().__init__(name)
        self._age = age

    @property
    def age(self):
        return self._age

    @age.setter
    def age(self, age):
        self._age = age

    def bark(self):
        print("%s岁的%s狗叫~~~"%(self.age, self.name))

    def run(self):
        print("狗跑~~~")

    def sleep(self):
        super().sleep()

d = Dog("旺财", 18)

d.sleep()
d.run()
d.bark()

print(isinstance(d, Dog))
print(isinstance(d, Animal))
print(issubclass(Dog, object))
