    # Service

    @Slf4j
@Service
public class WxLoginServiceImpl implements WxLoginService {
    @Autowired
    MemberService memberService;
    @Autowired
    WxLoginProperties wxLoginProperties;
    @Override
    public String wxLogin(HttpSession session) {
        try {
            //为浏览器重定向响应一个wx授权服务器的连接地址，访问时可以获取到wx登录二维码
            String url = "https://open.weixin.qq.com/connect/qrconnect?" +
                    "appid=%s" + //appid 表示当前应用在wx平台注册时的标识
                    "&redirect_uri=%s" + //wx用户扫码授权后，wx平台让浏览器重定向访问的地址(回调接口)
                    "&response_type=code" +
                    "&scope=snsapi_login" + // 用户授权后的code的作用域，可以获取用户的哪些信息
                    "&state=%s" +  //为了保证 将来调用回调接口的请求和访问当前接口的请求是同一个，使用state进行校验
                    "#wechat_redirect";
            //当前服务器中的一个接口地址，用户授权后 浏览器会自动访问该接口
            // 域名+端口号  必须和 应用注册在wx开放平台中的 授权回调域一样
            String redirectUrl = wxLoginProperties.getRedirectUri();
            String state = UUID.randomUUID().toString().replace("-" , "");
            session.setAttribute("state" ,state);
            int i = 1/0;
            url = String.format(url, wxLoginProperties.getAppId(),
                    URLEncoder.encode(redirectUrl, "UTF-8"), // URLEncoder可以将字符串中的特殊符号或中文转为unicode编码的字符
                    state);
            return "redirect:"+url;
        } catch (Exception e) {
//            throw new GuliException(ResultCodeEnum.UNKNOWN_REASON);
            return "redirect:http://localhost:3000/error"; //前端项目的错误页面地址
        }
    }

    @Override
    public String callback(String code, String state, HttpSession session) {
        //验证state的值，防止恶意攻击
        Object sessionState = session.getAttribute("state");
        if(sessionState==null ||
                StringUtils.isEmpty(state)||
                !sessionState.equals(state)
        ){
            throw new GuliException(ResultCodeEnum.UNKNOWN_REASON);
            //            return "redirect:"+ 前端项目的错误页面地址
        }
        session.removeAttribute("state");

        //2、根据code获取授权的accessToken
        try {
            String url = "https://api.weixin.qq.com/sns/oauth2/access_token?" +
                    "appid="+wxLoginProperties.getAppId() +
                    "&secret="+wxLoginProperties.getAppSecret() +
                    "&code="+ code +
                    "&grant_type=authorization_code";

            //初始化客户端对象
            HttpClientUtils client = new HttpClientUtils(url);
            //发起请求
            client.get();
            //获取响应体内容
            String content = client.getContent();
            System.out.println(content);
            Gson gson = new Gson();
            Map map = gson.fromJson(content, Map.class);
            if(map.get("errcode")!=null){
                log.error(content);
                throw new GuliException(ResultCodeEnum.UNKNOWN_REASON);
            }
            String accessToken = map.get("access_token").toString();//授权的token
            String openid = map.get("openid").toString();//wx用户的唯一标识

            //优化===  如果openid的用户第一次通过wx访问，查询wx用户数据后将他转为自己的用户存到数据库，如果是第N次访问，直接返回数据库数据
            LambdaQueryWrapper<Member> queryWrapper = new LambdaQueryWrapper<>();
            queryWrapper.eq(Member::getOpenid , openid);
            Member member = memberService.getOne(queryWrapper);
            if(member==null){//第一次wx登录访问  查询wx用户数据
                //3、根据accessToken获取微信用户的信息
                url = "https://api.weixin.qq.com/sns/userinfo?" +
                        "access_token="+ accessToken +
                        "&openid="+ openid;
                client = new HttpClientUtils(url);
                client.get();
                content = client.getContent();
                map = gson.fromJson(content , Map.class);
                if(map.get("errcode")!=null){
                    log.error(content);
                    throw new GuliException(ResultCodeEnum.UNKNOWN_REASON);
                }
                String nickname = map.get("nickname").toString();
                Integer sex = (int)(Double.parseDouble(map.get("sex").toString()));
                String province = map.get("province").toString();
                String headimgurl = map.get("headimgurl").toString();
                //存到数据库
                member = new Member();
                member.setAvatar(headimgurl);
                member.setSex(sex);
                member.setNickname(nickname);
                member.setOpenid(openid);
                System.out.println(content);
                memberService.save(member);

            }else if((System.currentTimeMillis()-member.getGmtModified().getTime())>=24*3*60*60*1000){//如果wx用户数据更新时间超过3天，重新查询wx用户的数据更新
                //3、根据accessToken获取微信用户的信息
                url = "https://api.weixin.qq.com/sns/userinfo?" +
                        "access_token="+ accessToken +
                        "&openid="+ openid;
                client = new HttpClientUtils(url);
                client.get();
                content = client.getContent();
                map = gson.fromJson(content , Map.class);
                if(map.get("errcode")!=null){
                    log.error(content);
                    throw new GuliException(ResultCodeEnum.UNKNOWN_REASON);
                }
                String nickname = map.get("nickname").toString();
                Integer sex = (int)(Double.parseDouble(map.get("sex").toString()));
                String province = map.get("province").toString();
                String headimgurl = map.get("headimgurl").toString();
                //存到数据库
                member.setAvatar(headimgurl);
                member.setSex(sex);
                member.setNickname(nickname);
                System.out.println(content);
                memberService.updateById(member);
            }
            //4、将wx用户数据转为jwt
            JwtInfo jwtInfo = new JwtInfo();
            jwtInfo.setNickname(member.getNickname());
            jwtInfo.setAvatar(member.getAvatar());
            jwtInfo.setId(member.getId());//id标识数据库用户的唯一的id
            String token = JwtHelper.createToken(jwtInfo);
            //跳转到登录成功页面  回显wx用户的信息
            return "redirect:http://localhost:3000?token="+token;
        } catch (Exception e) {
            throw new GuliException(ResultCodeEnum.UNKNOWN_REASON);
        }
    }
}

    # 前端

    mounted() {
    // 判断 如果路径中包含 token参数，代表用户是wx登录成功后跳转回当前页面
    if (this.$route.query.token) {
      local.save('GULI-TOKEN', this.$route.query.token)
      window.location = '/'
    }
	const guliUser = local.get('GULI-USER')
    if (guliUser) {
      this.user = JSON.parse(guliUser) // 需要将json字符串转为json对象
      return
    }
    // 从localStorage中获取GULI-TOKEN， 然后发送请求判断GULI-TOKEN是否有效，查询用户信息
    const guliToken = local.get('GULI-TOKEN')
    if (guliToken) {
      ucenterApi.getUserInfo(guliToken).then(r => {
        this.user = r.data.item
        console.log(this.user)
        // 查询到用户信息后可以缓存到浏览器中，避免刷新页面再次查询
        local.save('GULI-USER', JSON.stringify(this.user)) // json 对象不能直接缓存，必须转为json字符串
      })
    }
},
methods: {
    logout() {
        local.remove('GULI-TOKEN')
        local.remove('GULI-USER')
    window.location = '/'
    }
}

    export default{
  // 新增数据
  save(k, v) {
    window.localStorage.setItem(k, v)
  },
  // 获取数据
  get(k) {
    return window.localStorage.getItem(k)
  },
  // 删除数据
  remove(k) {
    window.localStorage.removeItem(k)
