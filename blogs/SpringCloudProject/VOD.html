    # 阿里云vod

    # java整合阿里云vod

    官方文档：https://help.aliyun.com/document_detail/53406.html

    上传SDK： 服务端SDK

    aliyun-java-vod-upload-1.4.14.jar： 下载的demo中有

    vod整合时需要使用的阿里云vod上传组件 jar包 没有开源，只能通过两种方式使用：

    ​ 1、导入jar包的方式

    ​ 2、将jar包安装到本地仓库

    mvn install:install-file -DgroupId=com.alibaba -DartifactId=aliyun-java-vod-upload -Dversion=1.4.14 -Dpackaging=jar -Dfile=aliyun-java-vod-upload-1.4.14.jar
    1

    安装时使用的路径拼接： com/alibaba/aliyun-java-vod-upload/1.4.14/aliyun-java-vod-upload-1.4.14.jar

    # 视频上传

    static String accessKeyId = "";
static String accessKeySecret = "";

//  普通上传
public static void mains(String[] args) {
    String title = "上传到阿里云后的视频名字";
    String fileName = "本地视频地址";

    UploadVideoRequest request = new UploadVideoRequest(accessKeyId,accessKeySecret,title,fileName);
    UploadVideoImpl uploader = new UploadVideoImpl();
    uploader.uploadVideo(request);
}

// 工作流上传
public static void mainw(String[] args) {
    String title = "上传到阿里云后的视频名字";
    String fileName = "本地视频地址";

    UploadVideoRequest request = new UploadVideoRequest(accessKeyId,accessKeySecret,title,fileName);
    // 设置工作流id
    request.setWorkflowId("9ebf53b185be1f55597538d97a1c1b36");
    UploadVideoImpl uploader = new UploadVideoImpl();
    uploader.uploadVideo(request);
}

// 文件流上传
public static void mainis(String[] args) throws FileNotFoundException {
    // 创建客户端对象
    UploadVideoImpl uploader = new UploadVideoImpl();
    // 配置请求参数
    String title = "上传到阿里云后的视频名字";
    String fileName = "本地视频地址";

    FileInputStream is = new FileInputStream(fileName);
    // 第一个 title 表示上传后的视频名称，可以没有扩展名
    // 第二个 title 表示上传时的视频名称，必须要扩展名
    UploadStreamRequest request = new UploadStreamRequest(accessKeyId,accessKeySecret,title,title,is);
    // 设置工作流id
    request.setWorkflowId("9ebf53b185be1f55597538d97a1c1b36");
    uploader.uploadStream(request);
}

    # 视频播放

    # 1.获取播放地址和播放凭证

    文档： https://help.aliyun.com/document_detail/61064.html

    web播放器文档：https://help.aliyun.com/document_detail/125547.html

    获取未加密视频播放地址:

    // 获取视频播放地址
public static void mainget(String[] argv) {
    // 1.创建客户端对象
    String regionId = "cn-shanghai";  // 点播服务接入区域
    DefaultProfile profile = DefaultProfile.getProfile(regionId,accessKeyId,accessKeySecret);
    DefaultAcsClient client = new DefaultAcsClient(profile);

    GetPlayInfoResponse response = new GetPlayInfoResponse();
    try {
        // 2.创建请求对象：配置要查询播放地址的视频id
        GetPlayInfoRequest request = new GetPlayInfoRequest();
        request.setVideoId("5f78d8c8b39e47269f9ae5e0b4de0173");

        response = client.getAcsResponse(request);

        List<GetPlayInfoResponse.PlayInfo> playInfoList = response.getPlayInfoList();
        //播放地址
        for (GetPlayInfoResponse.PlayInfo playInfo : playInfoList) {
            System.out.print("PlayInfo.PlayURL = " + playInfo.getPlayURL() + "\n");
        }
        //Base信息
        System.out.print("VideoBase.Title = " + response.getVideoBase().getTitle() + "\n");
    } catch (Exception e) {
        System.out.print("ErrorMessage = " + e.getLocalizedMessage());
    }
    System.out.print("RequestId = " + response.getRequestId() + "\n");
}

    获取加密视频的播放凭证

    // 获取加密过的视频的 播放凭证
public static void main(String[] argv) {
    // 1.创建客户端对象
    String regionId = "cn-shanghai";  // 点播服务接入区域
    DefaultProfile profile = DefaultProfile.getProfile(regionId,accessKeyId,accessKeySecret);

    DefaultAcsClient client = new DefaultAcsClient(profile);
    GetVideoPlayAuthResponse response = new GetVideoPlayAuthResponse();
    try {
        GetVideoPlayAuthRequest request = new GetVideoPlayAuthRequest();
        request.setVideoId("48168c3c527540338cb3a5fce8ca7268");

        response = client.getAcsResponse(request);
        //播放凭证
        System.out.print("PlayAuth = " + response.getPlayAuth() + "\n");
        //VideoMeta信息
        System.out.print("VideoMeta.Title = " + response.getVideoMeta().getTitle() + "\n");
    } catch (Exception e) {
        System.out.print("ErrorMessage = " + e.getLocalizedMessage());
    }
    System.out.print("RequestId = " + response.getRequestId() + "\n");
}
