    excel处理

    Awaken 7/5/2021 easyexcel

    # easyexcel

    apache poi: 一次性将整个excel文档中的数据加载到内存中 再进行使用

    ​ 占内存

    easyexcel: 一次只加载一行数据到内存中 进行使用

    ​ 节省内存

    创建bean对应excel的字段

    @Data
@AllArgsConstructor
@NoArgsConstructor
public class Stus {
    @ExcelProperty(value = "编号" , index = 8) // index表示插入到excel中的索引位置，从0开始
    private String id;
    @ExcelProperty(value = "年龄") //不设置index表示从左往右匹配索引插入
    private Integer age;
    @ExcelProperty(value = "账号")
    private String username;
    @ExcelProperty(value = "密码")
    @ExcelIgnore //写数据时忽略该属性
    private String password;
    @ExcelProperty("身高")
    @NumberFormat("#.##") //保留小数点后两位
    private Double height;
    @ExcelProperty("生日")
    @DateTimeFormat("yyyy年MM月dd日") //日期格式化
    private Date birthday;
}

    @Test
public void testRead(){
    EasyExcel.read("E:\\API\\stus.xlsx")
        .head(Stus.class)  //如何解析excel，指定表头所对应的类
        .registerReadListener(new StusDataListener())
        //注册监听器：easyexcel每解析一行数据就会调用一次监听器 处理数据的业务方法
        .sheet(0)
        .doRead();
}

    public class StusDataListener extends AnalysisEventListener<Stus> {

    private List<Stus> data = new ArrayList<>();
    private int limit = 1000;

    @Override
    public void invoke(Stus stus, AnalysisContext analysisContext) {
        //保存数据到数据库中
        data.add(stus);
        if (data.size()==limit){
            System.out.println("invoke 开始保存数据");
            //清空data集合
            data.clear();
        }
    }

    @Override
    public void doAfterAllAnalysed(AnalysisContext analysisContext) {
        //将剩余数据保存数据库
        if (data.size()>0){
            System.out.println(data);
            System.out.println("doAfterAllAnalysed 开始保存数据");
            data.clear();
        }
    }
}
